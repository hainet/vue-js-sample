Reactive property

{{ value }}

> data.value // -> message
> vm.value // -> message

> data.value = 'updated'
> data.value // -> updated
> vm.value // -> updated

> vm.value = 'updated'
> data.value // -> updated
> vm.value // -> updated

> data = { value: 'updated' } // -> Out of Vue instance
> data.value // -> updated
> vm.value // -> message
